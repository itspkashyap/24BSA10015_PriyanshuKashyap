	public void read_data(String sl_no) {
		DB_Connection obj_DB_Connection = new DB_Connection();
		Connection connection = obj_DB_Connection.get_connection();
		PreparedStatement ps = null;
		ResultSet rs = null;
		try {
			String query = "SELECT * FROM user WHERE sl_no = ?";
			ps = connection.prepareStatement(query);
			ps.setString(1, sl_no);
			rs = ps.executeQuery();
			
			boolean hasRows = false;
			while (rs.next()) {
				hasRows = true;
				System.out.println("Sl no - " + rs.getString("sl_no"));
				System.out.println("Name  - " + rs.getString("name"));
				System.out.println("Email - " + rs.getString("email"));
			}
			if (!hasRows) {
				System.out.println("No record found matching sl_no: " + sl_no);
			}
		} catch (Exception e) {
			System.out.println(e);
		} finally {
			try { if (rs != null) rs.close(); } catch (Exception e) {}
			try { if (ps != null) ps.close(); } catch (Exception e) {}
			try { if (connection != null) connection.close(); } catch (Exception e) {}
		}
	}

	
	public void update_data(String sl_no, String new_sl_no, String name, String email) {
		DB_Connection obj_DB_Connection = new DB_Connection();
		Connection connection = obj_DB_Connection.get_connection();
		PreparedStatement ps = null;
		try {
			String query = "UPDATE user SET sl_no = ?, name = ?, email = ? WHERE sl_no = ?";
			ps = connection.prepareStatement(query);

			ps.setString(2, name);
			ps.setString(3, email);
			ps.setString(4, sl_no);
			ps.executeUpdate();
			System.out.println("Row updated successfully.");
		} catch (Exception e) {
			System.out.println(e);
		} finally {
			try { if (ps != null) ps.close(); } catch (Exception e) {}
			try { if (connection != null) connection.close(); } catch (Exception e) {}
		}
	}

	
	public void alter_table_structure(String columnName, String columnType) {
		DB_Connection obj_DB_Connection = new DB_Connection();
		Connection connection = obj_DB_Connection.get_connection();
		PreparedStatement checkPs = null;
		PreparedStatement alterPs = null;
		ResultSet rs = null;
		try {
			String checkQuery = "SELECT COUNT(*) FROM INFORMATION_SCHEMA.COLUMNS " +
			                    "WHERE TABLE_NAME = 'user' AND COLUMN_NAME = ?";
			checkPs = connection.prepareStatement(checkQuery);
			checkPs.setString(1, columnName);
			rs = checkPs.executeQuery();
			
			if (rs.next() && rs.getInt(1) == 0) {
				String alterQuery = "ALTER TABLE user ADD COLUMN " + columnName + " " + columnType;
				alterPs = connection.prepareStatement(alterQuery);
				alterPs.executeUpdate();
				System.out.println("Table structure altered. Column added: " + columnName);
			} else {
				System.out.println("Column '" + columnName + "' already exists in layout structure.");
			}
		} catch (Exception e) {
			System.out.println(e);
		} finally {
			try { if (rs != null) rs.close(); } catch (Exception e) {}
			try { if (checkPs != null) checkPs.close(); } catch (Exception e) {}
			try { if (alterPs != null) alterPs.close(); } catch (Exception e) {}
			try { if (connection != null) connection.close(); } catch (Exception e) {}
		}
	}

	// D - STANDARD DELETE (Deletes specific targeted rows)
	public void delete_data(String sl_no) {
		DB_Connection obj_DB_Connection = new DB_Connection();

		PreparedStatement ps = null;
		try {

			System.out.println("Deleted row with ID: " + sl_no);
		} catch (Exception e) {
			System.out.println(e);
		} finally {
			try { if (ps != null) ps.close(); } catch (Exception e) {}
			try { if (connection != null) connection.close(); } catch (Exception e) {}
		}
	}

	// DELETE TRUNCATE - Total quick cleanup wipeout
	public void truncate_table() {
		DB_Connection obj_DB_Connection = new DB_Connection();
		Connection connection = obj_DB_Connection.get_connection();
		PreparedStatement ps = null;
		try {
			String query = "TRUNCATE TABLE user";
			ps = connection.prepareStatement(query);
			ps.executeUpdate();
			System.out.println("Table wiped clean using TRUNCATE.");
		} catch (Exception e) {
			System.out.println(e);
		} finally {
			try { if (ps != null) ps.close(); } catch (Exception e) {}
			try { if (connection != null) connection.close(); } catch (Exception e) {}
		}
	}
}
