#Group by-
SELECT DEPARTMENT, COUNT(WORKER_ID)
FROM Worker
WHERE DEPARTMENT = 'Admin' OR DEPARTMENT = 'HR'
GROUP BY DEPARTMENT;

-- Count employees department wise
SELECT DEPARTMENT, COUNT(DEPARTMENT)
FROM Worker
GROUP BY DEPARTMENT;

-- Sum of salary department wise
SELECT DEPARTMENT, SUM(SALARY)
FROM Worker
GROUP BY DEPARTMENT;

-- HAVING
select FIRST_NAME ,DEPARTMENT ,count(DEPARTMENT) as count  from worker  group by DEPARTMENT having count(DEPARTMENT)>=3 order by DEPARTMENT  asc;
SELECT DEPARTMENT, SUM(SALARY) AS sum
FROM Worker
GROUP BY DEPARTMENT
HAVING SUM(SALARY) > 100000
ORDER BY sum ASC;
